	{
		for (;;)
		{
			c1 = rand() % (max - min) + min + 1;
			c2 = rand() % (max - min) + min + 1;
			int c3 = c1*c2;
			do
			{
				c3 = c3 - c1;

			} while (c3 > max);
			if ((c3 < max) && (c1 != 0))
			{
				cout << c3 << "/" << c1 << "=" << endl; 
				x = c3/c1;
				cout << "请输入你的答案：" << endl;
				cin >> z;
				if (z == x)
				{
					cout << "正确" << endl;
					++zq;
				}
				else
					cout << "错误，正确答案是：" << x << endl; break;
			}
		}
	}
}

				}
			}
			else
			{
				c2 = rand() % c1;
				if (c2 != 0)
				{
					cout << c1 << "/" << c2 << "=" << endl; 
					x = c1 / c2;
					int y = c1%c2;
					cout << "请输入你的答案：" << endl;
					int m;
					cin >> z>>m;
					if ((z == x)&&(y==m))
					{
						cout << "正确" << endl;
						++zq;
					}
					else
						cout << "错误，正确答案是：" << "(" << c1 << "/" << c2 << ")" << endl; break;
				}
			}
		}
	}
}
